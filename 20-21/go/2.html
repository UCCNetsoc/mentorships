# Lesson 2 Structs and interfaces (with examples)
# Structs Structs are representations of objects. They can have fields and methods (both private and public). ```go type Example struct { name string // lowercase means private Game int // uppercase means public } ```
# Struct methods ```go func (e *Example) Method1() { e.name = "bruh" } func (e Example) GetName() string { return e.name }
# Interfaces ```go type ExampleInterface interface { GetName string } type Employee struct { name, position string } func (e Employee) GetName() string { return e.name } type Manager struct { name, nickname string age int } func (e Manager) GetName() string { return e.name } func useIt(eg ExampleInterface) { name := e.GetName fmt.Println(name) } func main() { useIt(Employee{name: "GAMER"}) useIt(Manager{name: "NOT GAMER"}) }
