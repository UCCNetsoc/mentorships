# Lesson 1 The basics
# Hello World ```go package main import "fmt" func main() { fmt.Println("hello world") } ```
# Packages The `import "fmt"` is importing a package. In this case it's `fmt`, part of the standard library. Packages provide reusable code for you to use in your programs. In this case `fmt` provides formatting functions such as `Println`.
# Variables ```go package main import "fmt" func main() { var a = "initial" fmt.Println(a) var b, c int = 1, 2 fmt.Println(b, c) var d = true fmt.Println(d) var e int fmt.Println(e) f := "apple" fmt.Println(f) } ```
# If statements ```go package main import "fmt" func main() { i := 1 if i > 0 { fmt.Println("yes") } } ```
# For loop ```go i := 1 for i <= 3 { fmt.Println(i) i = i + 1 } for j := 7; j <= 9; j++ { fmt.Println(j) } for { fmt.Println("loop") break } for n := 0; n <= 5; n++ { if n%2 == 0 { continue } fmt.Println(n) } ```
# Binary Literals ```go package main import "fmt" func main() { a := 1 c := 0b1 fmt.Println(a == c) } ```
# Arrays ```go package main import "fmt" func main() { a := [10]int{1, 2, 3, 4, 5} for _, value := range a { fmt.Println(value) } } ```
# Slices ```go package main import "fmt" func main() { a := []int{1, 2, 3} fmt.Println(len(a), cap(a)) a = append(a, 1) // Copy O(N) fmt.Println(len(a), cap(a)) a = append(a, 1) // No Copy O(1) fmt.Println(len(a), cap(a)) } ```
# Switch ```go i := 2 fmt.Print("Write ", i, " as ") switch i { case 1: fmt.Println("one") case 2: fmt.Println("two") case 3: fmt.Println("three") } t := time.Now() switch { case t.Hour() < 12: fmt.Println("It's before noon") default: fmt.Println("It's after noon") } ```
# Type Switch ```go package main import "fmt" type gamer int func main() { var g gamer = 1 var i interface{} = g // i = 1 switch i.(type) { case gamer: fmt.Println("ITS A GAMER") case int: fmt.Println("ITS A INT") case string: fmt.Println("ITS A STRING") } } ```
